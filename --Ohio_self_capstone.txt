-- Initial Query. Looking at Hurricanes in North Atlantic to see if we can match hurricanes to weather patterns observed in Ohio
SELECT
    sid,
    season,
    basin,
    subbasin,
    iso_time
FROM `bigquery-public-data.noaa_hurricanes.hurricanes` 
WHERE basin = "NA" AND EXTRACT(YEAR FROM iso_time) > 2015
Order By iso_time desc
-- needed to isolate storms powerful enough to reach Ohio from the coast. Cutting to just ET- Extratropical and TS Tropical Storm
SELECT
    sid,
    season,
    basin,
    subbasin,
    iso_time
FROM `bigquery-public-data.noaa_hurricanes.hurricanes` 
WHERE basin = "NA" 
AND EXTRACT(YEAR FROM iso_time) > 2015 
AND nature IN ('TS', 'ET')
Order By iso_time desc
--Now to isolate sid numbers so I can see an idea of the amount of storms during that period.
SELECT
    sid,
    max (season) AS Season,
    max (subbasin) as SB
FROM `bigquery-public-data.noaa_hurricanes.hurricanes` 
WHERE basin = "NA" 
AND EXTRACT(YEAR FROM iso_time) >= 2015 
AND nature IN ('TS', 'ET')
Group BY sid
Order By season desc
-- I now am going to pivot with the idea to figure out how many of these storms on average hit the target basin
SELECT
  COUNT(distinct sid) / Count (distinct season) AS Avg_storms_per_season
FROM `bigquery-public-data.noaa_hurricanes.hurricanes` 
WHERE basin = "NA" 
AND EXTRACT(YEAR FROM iso_time) >= 2015 
AND nature IN ('TS', 'ET')
-- 18.8 storms per season in the NA Basin. Next step is to isolate to some coordinates that give it a reasonable chance to hit Ohio.
-- Shifting gears. Above didn't seem to give me what I wanted. So I decided rather than subdivide on subbasin I can grab a coordinate box around the gulf of Mexico. These give high probability of making weather changes in Ohio. So I will filter using the coordinate box. And then isolate Months of Sept- Dec which is Hunting Season. I am ignoring the months of January and Feb as these have low risk for storm, and are harder months to hunt so low odds of being able to sell a high value hunt here. --http://bboxfinder.com/#25.522615,-97.250977,30.259067,-81.782227
SELECT
    sid,
    Season,
    iso_time,
    latitude,
    longitude,
    nature
FROM `bigquery-public-data.noaa_hurricanes.hurricanes` 
WHERE longitude <= -81 AND longitude >= -97.5 
AND latitude  >= 24.5 AND latitude <= 30.5
AND EXTRACT(YEAR FROM iso_time) >= 2015 
AND Extract(MONTH FROM iso_time) >= 9
AND nature IN ('TS', 'ET')
Order By season desc
-- Above gave me 246 instances of storm catching. I could further aggregate the SID numbers to get an exact amount but this feels sufficient for now. 
-- Shifting focus no to sever storms to see if these can have an effect on hunting season weather. Ohio Archery Season opens on the 4th Weekend of September. Making earliest possible date the 22nd. Will filter to see if a storm is possible. 
SELECT  
  *
FROM `bigquery-public-data.noaa_historic_severe_storms.storms_*` 
WHERE _TABLE_SUFFIX BETWEEN '2015' AND '2025'
AND state = 'Oh'
AND 
  (
    EXTRACT(MONTH FROM event_begin_time) =9 AND EXTRACT(DAY FROM event_begin_time) >= 22
    OR EXTRACT(MONTH FROM event_begin_time) BETWEEN 10 AND 11
    OR EXTRACT(MONTH FROM event_begin_time) = 12 AND EXTRACT(DAY FROM event_begin_time) <= 31
  )
ORDER BY EXTRACT(YEAR FROM event_begin_time) DESC
-- BASED on these results would say it is a major factor to consider. 

-- ON to Deer Harvest.. Cleaned prepared my data from the ODNR Weekly Harvest Reports. 
WITH weekly_totals AS (
  SELECT
    period_end_date,
    SUM(Antlered) AS weekly_bucks,
    SUM(Antlerless) AS weekly_does,
    SUM(Antlered + Antlerless) AS weekly_total
  FROM `project-7f368563-dc5e-4ab9-af8.OH_Hunting.OH_24_weekly_harvest`
  GROUP BY period_end_date
)

SELECT
  AVG(weekly_bucks) AS avg_bucks_per_week,
  AVG(weekly_does) AS avg_does_per_week,
  AVG(weekly_total) AS avg_total_per_week
FROM weekly_totals;
-- SO this gives me a weekly average of animals harvested per week. Goal being I can now put weeks in buckets. 
-- Now I have a second year of weekly of data to play with. 
SELECT
    SUM(Antlered) AS Bucks,
    SUM(Antlerless) AS Does,
    period_end_date
   
FROM (
    SELECT Antlered, Antlerless, period_end_date, Harvest_method
    FROM `project-7f368563-dc5e-4ab9-af8.OH_Hunting.OH_23_Weekly_Harvest`

    UNION ALL

    SELECT Antlered, Antlerless, period_end_date, Harvest_method
    FROM `project-7f368563-dc5e-4ab9-af8.OH_Hunting.OH_24_weekly_harvest`
)
GROUP BY period_end_date
ORDER BY Bucks DESC
